\section{Patterns}
\label{sec:patterns}

\syntax\begin{lstlisting}
Pattern        ::= Pattern1 {'|' Pattern1}
Pattern1       ::= variable_id ':' Type_Pat
                 | '_' ':' Type_Pat
                 | Pattern2
Pattern2       ::= variable_id ['@' Simple_Pattern]
                 | Simple_Pattern
Simple_Pattern ::= '_'
                 | variable_id
                 | literal
                 | Stable_Id
                 | Stable_Id '(' [Patterns] ')' 
                 | '(' [Patterns] ')'
Patterns       ::= Pattern {',' Pattern}
\end{lstlisting}
